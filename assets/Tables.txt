Table Accounts {
  id integer [primary key]
  login varchar
  passwordHash varchar
  refreshToken varchar
  tokenCreated varchar
  tokenExpires varchar
}


Table Tasks {
  id integer [primary key]
  accountId integer
  content string
  isUrgent bool
  isImportant bool
  deadline datetime
  isCompleted bool
}

Table Wishes {
  wishId integer [primary key]
  accountId integer
  content string
  isCompleted bool
}

Table GHabits {
  id int [primary key]
  accountId int
  content string
}


Table GHabitDate {
id int [primary key]
ghabitId int
date datetime
}

Table BHabits {
  id int [primary key]
accountId int
content string
issueDate date
}

Table BHabitDate {
  id int [primary key]
bhabitId int
date datetime
}

Table Quotes {
id int [primary key]
author string
quote string
}

Table UserQuotes {
  id int
  accountId int
  string quote
}

Ref: Accounts.id < BHabits.accountId
Ref: Accounts.id < UserQuotes.accountId
Ref: GHabits.id < GHabitDate.ghabitId
Ref: BHabits.id < BHabitDate.bhabitId
